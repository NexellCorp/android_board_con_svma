# Before fusing, partmap must be flashed to target device
# Target device uses this partmap information to fuse flash images by fastboot

flash=mmc,0:boot_a:emmc:0x500000,0x4000000:boot_a.img;
flash=mmc,0:boot_b:emmc:0x4600000,0x4000000:boot_b.img;
flash=mmc,0:system:ext4:0x8700000,0x40000000:system.img;
flash=mmc,0:vendor:ext4:0x48800000,0x10000000:vendor.img;
flash=mmc,0:userdata:ext4:0x58900000,0:userdata.img;
